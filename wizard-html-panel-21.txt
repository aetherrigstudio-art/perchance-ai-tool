  function getSection(label, text){
    var header = "=== " + label + " ===";
    var i = text.indexOf(header);
    if(i < 0){ console.warn("[wizard] getSection: label not found:", label); return ""; }

    if(n >= 0) rest = rest.slice(0, n);

  }
  function replaceSection(full, label, newBody){
    var header = "=== " + label + " ===";
    newBody = (newBody || "").trim();
    var i = full.indexOf(header);
    if(i < 0) return full.trim() + "\n\n" + header + "\n" + newBody + "\n";

    if(afterHeader < 0) return full + "\n" + newBody;
    var rest = full.slice(afterHeader + 1);
    var n = rest.search(/\n===\s+[A-Z]/);
    var tail = (n >= 0) ? rest.slice(n) : "";
    return full.slice(0, afterHeader + 1) + newBody + (tail || "\n");
  }

  function escHtml(s){ return (s||"").replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;"); }
  function escAttr(s){ return escHtml(s).replace(/"/g,"&quot;").replace(/'/g,"&#39;"); }
  function parseConsistency(raw){
    var out = [];
    raw.split(/\n(?=ISSUE:)/).forEach(function(b){
      var issue = (b.match(/ISSUE:\s*(.*)/) || [])[1];
      var ch = (b.match(/CHARACTER:\s*(.*)/) || [])[1];
      var fi = b.indexOf("FIX:");
      var fix = fi >= 0 ? b.slice(fi + 4).trim() : "";
      if(issue && fix) out.push({ issue: issue.trim(), character: (ch || "").trim(), fix: fix, applied: false });
    });
    return out;
  }
  function resolveCharBox(name){
    name = (name || "").trim().toLowerCase();
    if(!name) return null;
    if(val("mainOut").trim() && parseChar(val("mainOut")).name.toLowerCase() === name) return { kind:"main" };
    if(val("personaOut").trim() && parseChar(val("personaOut")).name.toLowerCase() === name) return { kind:"persona" };
    syncExtrasFromDOM();
    for(var i=0; i<extras.length; i++){ if((extras[i].out||"").trim() && parseChar(extras[i].out).name.toLowerCase() === name) return { kind:"extra", index:i }; }
    return null;
  }
  window.applyFix = function(i){
    var it = consistencyIssues[i]; if(!it) return;

    if(!box){ alert("Couldn't match \"" + it.character + "\" to a current character. You can edit that character manually using this fix:\n\n" + it.fix); return; }
    it.applied = true; renderConsistency();
    if(box.kind === "main"){ setVal("mainAdj", it.fix); genChar("main", true); }
    else if(box.kind === "persona"){ setVal("personaAdj", it.fix); genChar("persona", true); }
    else { setVal("exAdj_" + box.index, it.fix); genExtra(box.index, true); }
  };

  // Inline CRC-32: Perchance blocks external modules, so no crypto import is possible.
  function crc32(bytes){
    if(!crc32.t){ crc32.t = []; for(var n=0;n<256;n++){ var c=n; for(var k=0;k<8;k++) c=(c&1)?(0xEDB88320^(c>>>1)):(c>>>1); crc32.t[n]=c>>>0; } }
    var crc = 0xFFFFFFFF;
    for(var i=0;i<bytes.length;i++) crc = crc32.t[(crc ^ bytes[i]) & 0xFF] ^ (crc >>> 8);
    return (crc ^ 0xFFFFFFFF) >>> 0;
  }
  function u32(n){ return [(n>>>24)&255,(n>>>16)&255,(n>>>8)&255,n&255]; }
  function pngChunk(type, dataBytes){
    var td = []; for(var i=0;i<4;i++) td.push(type.charCodeAt(i));
    for(var j=0;j<dataBytes.length;j++) td.push(dataBytes[j]);
    var crc = crc32(new Uint8Array(td));
    return new Uint8Array([].concat(u32(dataBytes.length), td, u32(crc)));
  }
  function textData(keyword, text){
    var a = []; var i;
    for(i=0;i<keyword.length;i++) a.push(keyword.charCodeAt(i)&255);
    a.push(0);
    for(i=0;i<text.length;i++) a.push(text.charCodeAt(i)&255);
    return a;
  }
  function pngInsertText(png, keyword, text){
    var chunk = pngChunk("tEXt", textData(keyword, text));
    var off = 8;
    while(off < png.length){
      var len = ((png[off]<<24)|(png[off+1]<<16)|(png[off+2]<<8)|png[off+3]) >>> 0;
      var type = String.fromCharCode(png[off+4],png[off+5],png[off+6],png[off+7]);
      if(type === "IEND"){
        var out = new Uint8Array(off + chunk.length + (png.length - off));
        out.set(png.subarray(0, off), 0);
        out.set(chunk, off);
        out.set(png.subarray(off), off + chunk.length);
        return out;
      }
      off += 12 + len;
    }
    return png;
  }

    var g = cv.getContext("2d");
    g.fillStyle = "#3a3f4b"; g.fillRect(0,0,512,768);
    g.fillStyle = "#cdd3df"; g.font = "bold 34px sans-serif"; g.textAlign = "center";
    g.fillText((name || "Character").slice(0,24), 256, 384);
    var b64 = cv.toDataURL("image/png").split(",")[1];
    var bin = atob(b64); var bytes = new Uint8Array(bin.length);
    for(var i=0;i<bin.length;i++) bytes[i] = bin.charCodeAt(i);
    cb(bytes);
  }
  function canvasToBytes(cv){
    var b64 = cv.toDataURL("image/png").split(",")[1];
    var bin = atob(b64); var bytes = new Uint8Array(bin.length);
    for(var i=0;i<bin.length;i++) bytes[i] = bin.charCodeAt(i);
    return bytes;
  }
  function makeCardFromImage(url, name, cb){
    if(!url){ makeBasePng(name, cb); return; }
    var img = new Image();
    img.crossOrigin = "anonymous";
    img.onload = function(){
      try{
        var cv = document.createElement("canvas"); cv.width = 512; cv.height = 768;
        var g = cv.getContext("2d");
        var car = 512/768, ar = img.width / img.height, sw, sh, sx, sy;
        if(ar > car){ sh = img.height; sw = sh * car; sx = (img.width - sw)/2; sy = 0; }
        else { sw = img.width; sh = sw / car; sx = 0; sy = (img.height - sh)/2; }
        g.drawImage(img, sx, sy, sw, sh, 0, 0, 512, 768);
        cb(canvasToBytes(cv));
      } catch(e){ makeBasePng(name, cb); }
    };
    img.onerror = function(){ makeBasePng(name, cb); };
    img.src = url;
  }
