async genCharacterImage(prompt, negativePrompt, resolution) =>
  let opts = { prompt: prompt, negativePrompt: (negativePrompt || ""), resolution: (resolution || "512x768") };
  let result = null;
  try { result = await imageGen(opts); }
  catch(e) { try { result = await imageGen.image(opts); } catch(e2) { return ""; } }
  if(!result) return "";
  if(typeof result === "string") return result;
  if(result.imageURL) return result.imageURL;
  if(result.imageUrl) return result.imageUrl;
  if(result.url) return result.url;
  if(result.src) return result.src;
  if(result.outputEl && result.outputEl.src) return result.outputEl.src;
  if(typeof result.querySelector === "function") { let im = result.querySelector("img"); if(im) return im.src; }
  return "";

// Fresh words from Perchance's shared word banks (adjective/noun/verb), used by
// the HTML to rotate in real vocabulary every 4 hours. Fully defensive: any
// import/selection failure returns "" so the generator keeps working.
wordBank(type, count) =>
  try {
    let n = count || 3;
    let list = (type == "noun") ? nounBank : ((type == "verb") ? verbBank : adjBank);
    if(!list) return "";
    let out = [];
    for(let i = 0; i < n; i++) {
      let w = (typeof list.selectOne === "function") ? list.selectOne() : list.selectOne;
      if(w) out.push(("" + w).trim());
    }
    return out.join("|");
  } catch(e) { return ""; }

// Upload a Blob (the gzipped share file) to Perchance's file store and return
// its URL, used by the HTML to build a one-click "open in chat" share link.
// Defensive: any failure returns "" or "ERR:<reason>" so the UI can report it.
async uploadShare(blob) =>
  try {
    let r = await uploader(blob);
    if(!r) return "";
    if(r.error) return "ERR:" + r.error;
    return r.url || "";
  } catch(e) { return "ERR:" + (e && e.message ? e.message : "upload failed"); }
