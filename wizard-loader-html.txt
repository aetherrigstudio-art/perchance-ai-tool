<div id="wiz-root">Loading the latest build…</div>

<script>
/* AI Character Set Builder - REMOTE LOADER.
 *
 * Paste this ONCE into the Perchance HTML editor. After that, every push to the
 * GitHub branch is live automatically - editing the repo IS deploying, so you
 * never copy-paste the HTML panel again.
 *
 * How it works: at runtime it fetches the real HTML panel (markup + <script> +
 * <style>) from GitHub raw, injects it, and re-executes its scripts (innerHTML
 * never runs <script>, so we re-create each one). Because the real markup is
 * injected by JS, Perchance never templates it - so the [ ] { } bracket rules
 * stop applying to it too.
 *
 * The data panel (char-wiz-dat) is still pasted once into the data/generator
 * editor; it rarely changes. Re-paste it only when char-wiz-dat itself changes.
 *
 * Requires: the source repo/file readable without auth (it is - public + CORS *).
 */
(function () {
  // The ONE line to change if you move the source (e.g. after merging to main,
  // point it at .../main/char-wiz-html instead of the branch):
  var SRC = "https://raw.githubusercontent.com/aetherrigstudio-art/perchance-ai-tool/claude/init-9i0np9/char-wiz-html";

  var root = document.getElementById("wiz-root");

  function fail(msg) {
    root.textContent =
      "Could not load the latest build (" + msg + "). As a fallback, open " +
      SRC + " and paste its contents here manually.";
  }

  fetch(SRC + "?t=" + Date.now(), { cache: "no-store" })
    .then(function (r) {
      if (!r.ok) throw new Error("HTTP " + r.status);
      return r.text();
    })
    .then(function (html) {
      root.innerHTML = html;                 // markup + <style> take effect now
      // innerHTML does not execute <script> nodes; re-create them in order so
      // the panel's logic (showTab, resetAll, load(), buildWizardPrompt, ...)
      // is defined on the global scope, exactly as a direct paste would do.
      var scripts = root.querySelectorAll("script");
      for (var i = 0; i < scripts.length; i++) {
        var old = scripts[i];
        var s = document.createElement("script");
        for (var j = 0; j < old.attributes.length; j++) {
          s.setAttribute(old.attributes[j].name, old.attributes[j].value);
        }
        s.textContent = old.textContent;
        old.parentNode.replaceChild(s, old);
      }
    })
    .catch(function (e) {
      fail(e && e.message ? e.message : "fetch blocked");
    });
})();
</script>
