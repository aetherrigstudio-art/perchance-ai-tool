    <div id="subBtns" class="btns"></div>
    <p class="hint" style="margin-top:0.75rem;">Optional extra notes (mood, palette, era):</p>
    <input id="styleNotes" class="adj" placeholder="e.g. muted teal palette, foggy, melancholic">
    <div class="row"><button onclick="genStyle(false);">build style</button></div>

    <p class="hint" style="margin-top:0.75rem;">Adjustment (regenerates both, keeping your base style):</p>

    <div class="row"><button onclick="genStyle(true);">apply adjustment</button></div>

<script>
  var STORE_KEY = "accStyleV1";
  var STYLES = {
    "Anime": ["modern anime", "retro 90s anime", "Ghibli-esque", "dark seinen", "soft cel shading", "vibrant colors", "manhwa webtoon"],
    "Photoreal": ["cinematic film still", "portrait photography", "35mm film grain", "golden hour", "studio lighting", "moody low-key", "hyperreal"],
    "Oil painting": ["classical", "impressionist", "baroque chiaroscuro", "alla prima", "thick impasto", "romanticism"],
    "Watercolor": ["soft wash", "ink and wash", "loose brushwork", "pastel palette", "storybook"],
    "Comic / ink": ["western comic", "manga ink", "film noir", "graphic novel", "bold line art", "halftone"],
    "Fantasy art": ["dark fantasy", "high fantasy", "painterly", "concept art", "matte painting", "ethereal"],
    "3D render": ["Pixar-style", "octane render", "Unreal Engine", "claymation", "stylized 3D", "soft global illumination"],
    "Retro / pixel": ["pixel art", "16-bit", "vaporwave", "synthwave", "80s airbrush", "VHS grain"]
  };

  var state = { main: "", subsByMain: {}, notes: "", prefix: "", suffix: "", raw: "" };
  try { var s = JSON.parse(localStorage.getItem(STORE_KEY)); if(s && typeof s === "object"){ for(var k in s) state[k] = s[k]; } } catch(e){}
  if(!state.subsByMain || typeof state.subsByMain !== "object") state.subsByMain = {};

  function el(id){ return document.getElementById(id); }
  function val(id){ var e = el(id); return e ? e.value : ""; }
  function setVal(id,v){ var e = el(id); if(e) e.value = (v==null?"":v); }
  function getSection(label, text){
    var header = "=== " + label + " ===";
    var i = text.indexOf(header); if(i<0) return "";
    var le = text.indexOf("\n", i); if(le<0) return "";
    var rest = text.slice(le+1);
    var n = rest.search(/\n===\s+[A-Z]/); if(n>=0) rest = rest.slice(0,n);
    return rest.trim();
  }

  function save(){
    state.notes = val("styleNotes"); state.prefix = val("prefixOut");
    state.suffix = val("suffixOut"); state.raw = val("styleRaw");
    try { localStorage.setItem(STORE_KEY, JSON.stringify(state)); } catch(e){}
  }

  function renderMains(){
    var c = el("mainBtns"); c.innerHTML = "";
    Object.keys(STYLES).forEach(function(name){
      var b = document.createElement("button");
      b.className = "styleBtn" + (state.main === name ? " active" : "");
      b.textContent = name;
      b.onclick = function(){ state.main = name; if(!state.subsByMain[name]) state.subsByMain[name] = []; save(); renderMains(); renderSubs(); updateStatus(); };
      c.appendChild(b);
    });
  }
  function renderSubs(){
    if(!state.main){ el("subCard").style.display = "none"; return; }
    el("subCard").style.display = "";
    var c = el("subBtns"); c.innerHTML = "";
    var sel = state.subsByMain[state.main] || [];
    STYLES[state.main].forEach(function(sub){
      var b = document.createElement("button");
      b.className = "styleBtn" + (sel.indexOf(sub) >= 0 ? " active" : "");
      b.textContent = sub;
      b.onclick = function(){
        var arr = state.subsByMain[state.main] || [];
        var i = arr.indexOf(sub);
        if(i >= 0) arr.splice(i,1); else arr.push(sub);
        state.subsByMain[state.main] = arr; save(); renderSubs(); updateStatus();
      };
      c.appendChild(b);
    });
  }
  function updateStatus(){
    if(!state.main){ el("status").textContent = "pick a base style"; return; }
    var subs = state.subsByMain[state.main] || [];
    el("status").textContent = state.main + (subs.length ? " + " + subs.join(", ") : "");
  }

  window.buildStylePrompt = function(){
    var subs = (state.subsByMain[state.main] || []).join(", ");
    var base = "You are an expert image-prompt engineer for a diffusion model used in a roleplay chat. Turn the chosen art style into reusable prompt fragments that are PREPENDED and APPENDED to EVERY image. Describe ONLY medium, art style, rendering technique, lighting and atmosphere. NEVER mention characters, subjects, poses, or specific backgrounds. Do NOT use any negativePrompt syntax.\n\n" +
      "BASE STYLE: " + state.main + "\nMODIFIERS: " + (subs || "(none)") + "\nEXTRA NOTES: " + (val("styleNotes") || "(none)") + "\n\n" +
      "Output exactly these two sections and nothing else:\n" +
      "=== PREFIX ===\nA comma-separated list of style / medium / lighting / atmosphere tags to place BEFORE every prompt. End with a trailing comma and a space. Keep it under about 12 tags.\n" +
      "=== SUFFIX ===\nA short comma-separated list of positive quality tags to place AFTER every prompt, beginning with a comma and a space (for example: , highly detailed, sharp focus, high quality). Under about 6 tags. No negative prompts.";
    if(window.refineMode){
      base += "\n\nCURRENT PREFIX:\n" + val("prefixOut") + "\n\nCURRENT SUFFIX:\n" + val("suffixOut") +
        "\n\nADJUSTMENT REQUESTED:\n" + (val("styleAdj") || "(just refine and tighten)") +
        "\n\nRewrite BOTH sections incorporating the adjustment. Keep the same two === labels and output only those.";
    }
    return base;
  };

  window.genStyle = function(refine){
    if(!state.main){ alert("Pick a base style first."); return; }
    window.activeSection = "style"; window.refineMode = !!refine;
    window.activeOutputEl = el("styleRaw");
    el("styleRaw").value = "";
    el("busyHint").textContent = "building style...";
    generate();
  };
  window.onStyleFinish = function(){
    el("busyHint").textContent = "";
    var raw = val("styleRaw");
    var pre = getSection("PREFIX", raw), suf = getSection("SUFFIX", raw);
    if(pre) setVal("prefixOut", pre);
    if(suf) setVal("suffixOut", suf);
    save();
  };

  window.copyField = function(id, noteId){
    var t = val(id);
    if(navigator.clipboard && navigator.clipboard.writeText){
      navigator.clipboard.writeText(t).then(function(){ el(noteId).textContent = "copied"; setTimeout(function(){ el(noteId).textContent = ""; }, 1500); });
    } else {
      var e = el(id); e.focus(); e.select(); el(noteId).textContent = "select + copy";
    }
  };

  window.resetAll = function(){
    if(!confirm("Clear the selected style and result?")) return;
    localStorage.removeItem(STORE_KEY);
    state = { main: "", subsByMain: {}, notes: "", prefix: "", suffix: "", raw: "" };
    ["styleNotes","prefixOut","suffixOut","styleRaw","styleAdj"].forEach(function(id){ setVal(id,""); });
    renderMains(); renderSubs(); updateStatus();
  };

  // restore
  setVal("styleNotes", state.notes); setVal("prefixOut", state.prefix);
  setVal("suffixOut", state.suffix); setVal("styleRaw", state.raw);
  ["styleNotes","prefixOut","suffixOut"].forEach(function(id){ var e = el(id); if(e) e.oninput = save; });
  renderMains(); renderSubs(); updateStatus();
</script>
